        incrementa quantidade de niveis da arvore

montaArvoreAritmetica(arvore, pilhaPolonesa)

    se pilha não vazia
        desempilha E
        cria no N = E
        se arvore estiver vazia
            arvore = N
            montaArvoreAritmetica(arvore, pilhaPolonesa)
        se arvore->esquerdo vazio
            arvore->esquerdo = N
            se E for operação
                montaArvoreAritmetica(N, pilhaPolonesa)
        senao
            se arvore->direito vazio
                arvore->direito = N
                se E for operacao
                    montaArvoreAritmetica(N, pilhaPolonesa)
        
        se arvore tiver ambos nos
            return
        senao 
            montaArvoreAritmetica(arvore, pilha)

montaEquacaoParentizada(arvore, pilha){
    se pilha vazia
        instancia pilha
    empilha ')'

    se os filhos de arvore forem folhas
        empilha arvore->direita
        empilha arvore
        empilha arvore->esquerda
    senao
        montaEquacaoParentizada(arvore->direita, pilha)
        empilha arvore
        montaEquacaoParentizada(arvore->esquerda, pilha)

    empilha '('
}
